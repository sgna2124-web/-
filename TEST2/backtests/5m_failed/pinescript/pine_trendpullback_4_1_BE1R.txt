//@version=5
strategy("OHLCV TrendPullback 4.1 (BE at 1R, Fixed TP)", overlay=true, initial_capital=10000, default_qty_type=strategy.percent_of_equity, default_qty_value=1, commission_type=strategy.commission.percent, commission_value=0.04, slippage=3, pyramiding=0, process_orders_on_close=true)

// NOTE: 이 파일은 1R 도달시 스탑을 진입가로 이동(1회)하는 BE 버전 기록용.
// 실제 테스트 결과 다수 종목에서 총수익을 깎는 경향으로 BE는 폐기 결론.

atrLen = input.int(14, "ATR 길이", minval=2)
atr = ta.atr(atrLen)

stopMult = input.float(1.6, "SL: ATR배", step=0.1)
rr = input.float(3.0, "RR: TP/SL", step=0.1)
useBE = input.bool(true, "BE: 1R 도달시 스탑=진입가(1회)")
beR = input.float(1.0, "BE 트리거(R)", step=0.1, minval=0.5)

var float entryPx = na
var float rDist = na
var float fixStop = na
var float fixTP = na
var bool beArmed = false

posNow = strategy.position_size
posPrev = strategy.position_size[1]
justEntered = (posNow != 0 and posPrev == 0)
justExited = (posNow == 0 and posPrev != 0)

if justExited
    entryPx := na
    rDist := na
    fixStop := na
    fixTP := na
    beArmed := false

// (나머지 진입/레짐/풀백 로직은 4.0과 동일하며, 저장소 용량상 여기서는 핵심 BE 로직만 보관)

if justEntered
    entryPx := strategy.position_avg_price
    rDist := stopMult * atr
    if posNow > 0
        fixStop := entryPx - rDist
        fixTP := entryPx + (rr * rDist)
    if posNow < 0
        fixStop := entryPx + rDist
        fixTP := entryPx - (rr * rDist)
    beArmed := false

if useBE and posNow > 0 and not beArmed and close >= entryPx + (beR * rDist)
    beArmed := true
if useBE and posNow < 0 and not beArmed and close <= entryPx - (beR * rDist)
    beArmed := true

stopToUse = (useBE and beArmed) ? entryPx : fixStop

if posNow > 0 and not na(stopToUse) and not na(fixTP)
    strategy.exit("XL", from_entry="L", stop=stopToUse, limit=fixTP)
if posNow < 0 and not na(stopToUse) and not na(fixTP)
    strategy.exit("XS", from_entry="S", stop=stopToUse, limit=fixTP)
